import pandas as pd

df = pd.DataFrame({"Name": ["Ana", "Ben", "Chad"], "Age": [23, 31, 27]})

new_row = pd.DataFrame([{"Name": "Dina", "Age": 29}])
df2 = pd.concat([df, new_row], ignore_index=True)
print("\nAfter append:\n", df2)

# Slice rows
print("\nSlice rows 1..2:\n", df2.iloc[1:3])

# Add another row via loc
df2.loc[len(df2)] = {"Name": "Evan", "Age": 33}
print("\nAfter adding one more row via loc:\n", df2)

# Delete rows by index
df3 = df2.drop(index=[1])

Compare Univariate and Bivariate analysis for the UCI diabetes dataset 
# 9. Compare Univariate vs Bivariate on Diabetes dataset (sklearn diabetes).
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.datasets import load_diabetes

diab = load_diabetes(as_frame=True)
df = diab.frame  # features + target

# Univariate: histograms and summary
df.drop(columns=["target"]).hist(figsize=(12, 8), bins=20)
plt.suptitle("Univariate: Feature Distributions", y=1.02)
plt.show()
print("Summary stats:\n", df.describe(numeric_only=True))

# Bivariate: correlation heatmap + target vs top features
corr = df.corr(numeric_only=True)
plt.figure(figsize=(8, 6))
sns.heatmap(corr, annot=False, cmap="coolwarm", center=0)
plt.title("Bivariate: Correlation Heatmap")
plt.show()

# Identify top 3 features by absolute correlation with target
top3 = corr["target"].drop("target").abs().sort_values(ascending=False).head(3).index.tolist()
print("Top 3 correlated with target:", top3)

# Scatter plots of target vs top features
plt.figure(figsize=(12, 3))
for i, col in enumerate(top3, 1):
    plt.subplot(1, 3, i)
    sns.regplot(x=df[col], y=df["target"], scatter_kws={"alpha": 0.6}, line_kws={"color": "orange"})
    plt.title(f"target vs {col}")
plt.tight_layout()
plt.show()
Observation: Univariate plots show each feature’s spread/skew. Bivariate plots/heatmap reveal which features are most predictive of the target (e.g., BMI and S5 often rank high for this dataset).

# 10. Load iris.csv, read DataFrame, filter species "Iris-setosa".
import pandas as pd

# Try local file, else fetch
try:
    iris = pd.read_csv("iris.csv")
except FileNotFoundError:
    iris = pd.read_csv("https://raw.githubusercontent.com/mwaskom/seaborn-data/master/iris.csv")

print("Head:\n", iris.head())

# Normalize species naming differences between datasets
species_col = "species"
unique_vals = iris[species_col].astype(str).unique()
target_label = "Iris-setosa" if any(v.startswith("Iris-") for v in unique_vals) else "setosa"

setosa = iris[iris[species_col].astype(str) == target_label]
print(f"\nRecords with species == '{target_label}':\n", setosa)

13.

Multiple Regression on a custom Car dataset (predict CO2 from Volume & Weight) 
# 13. Multiple Regression: predict CO2 using Volume & Weight.
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression

cars = pd.DataFrame({
    "Company": ["Ford", "Toyota", "VW", "BMW", "Honda", "Hyundai", "Tesla", "Peugeot"],
    "Model":   ["Fiesta", "Corolla", "Golf", "3-Series", "Civic", "i20", "Model 3", "208"],
    "Volume":  [1100, 1300, 1400, 1600, 1500, 1200, 0, 1200],
    "Weight":  [980, 1020, 1160, 1350, 1200, 980, 1620, 1040],
    "CO2":     [99, 95, 118, 135, 110, 102, 0, 109]
})

X = cars[["Volume", "Weight"]]
y = cars["CO2"]

X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=42, test_size=0.25)
model = LinearRegression()
model.fit(X_train, y_train)

print("Coefficients (Volume, Weight):", model.coef_)
print("Intercept:", model.intercept_)
print("R^2 (train):", model.score(X_train, y_train))
print("R^2 (test):", model.score(X_test, y_test))

# Predict a new sample
sample = pd.DataFrame({"Volume": [1500], "Weight": [1200]})
print("Predicted CO2 for", sample.to_dict(orient="records")[0], "=>", model.predict(sample)[0])
Note: Tesla's CO2 is set to 0 to reflect EV; in practice you might exclude EVs or model separately.

# 14.i. Normal distribution math with NumPy and histogram plot.
import numpy as np
import matplotlib.pyplot as plt

mu, sigma = 0, 1
x = np.linspace(mu - 4*sigma, mu + 4*sigma, 1000)
pdf = (1/(sigma*np.sqrt(2*np.pi))) * np.exp(-0.5*((x-mu)/sigma)**2)

samples = np.random.normal(mu, sigma, size=5000)

plt.figure(figsize=(8,4))
plt.hist(samples, bins=40, density=True, alpha=0.5, label="Histogram")
plt.plot(x, pdf, color="red", lw=2, label="PDF")
plt.title("Normal Distribution (mu=0, sigma=1)")
plt.legend()
plt.show()
ii. Visualize a contour plot using plt.contour, contourf, imshow, colorbar, clabel 
# 14.ii. Contour visualizations.
import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(-3, 3, 200)
y = np.linspace(-3, 3, 200)
X, Y = np.meshgrid(x, y)
Z = np.sin(X) * np.cos(Y)

fig, axs = plt.subplots(1, 3, figsize=(12, 3.6))

cs = axs[0].contour(X, Y, Z, levels=10, colors="k")
axs[0].clabel(cs, inline=True, fontsize=8)
axs[0].set_title("plt.contour + clabel")

cf = axs[1].contourf(X, Y, Z, levels=20, cmap="viridis")
plt.colorbar(cf, ax=axs[1], shrink=0.8)
axs[1].set_title("plt.contourf + colorbar")

im = axs[2].imshow(Z, origin="lower", extent=[x.min(), x.max(), y.min(), y.max()], cmap="magma", aspect="auto")
plt.colorbar(im, ax=axs[2], shrink=0.8)
axs[2].set_title("plt.imshow + colorbar")

plt.tight_layout()
plt.show()

# 17. Basemap procedure and example.
# 1) Install:
#    pip install basemap basemap-data
# 2) Import and create a map, draw coastlines, countries, parallels/meridians.
# 3) Convert lon/lat to map coords, plot markers/paths, show.

from mpl_toolkits.basemap import Basemap
import matplotlib.pyplot as plt

fig = plt.figure(figsize=(8,5))
m = Basemap(projection="merc", llcrnrlat=-60, urcrnrlat=80,
            llcrnrlon=-180, urcrnrlon=180, resolution="c")
m.drawcoastlines()
m.drawcountries()
m.fillcontinents(color="#e0e0e0", lake_color="#a4cafe")
m.drawmapboundary(fill_color="#a4cafe")
m.drawparallels(range(-60, 81, 30), labels=[1,0,0,0], fontsize=8)
m.drawmeridians(range(-180, 181, 60), labels=[0,0,0,1], fontsize=8)

# Plot a few city locations
lats = [37.7749, 51.5074, 28.6139]
lons = [-122.4194, -0.1278, 77.2090]
x, y = m(lons, lats)
m.scatter(x, y, marker="o", color="red", zorder=5)
for xi, yi, name in zip(x, y, ["San Francisco", "London", "New Delhi"]):
    plt.text(xi+200000, yi+200000, name, fontsize=8)

plt.title("Basemap Example - World Map")
plt.show()

Bivariate analysis: Hours spent studying vs Exam score (20 students) 
# 20. Bivariate analysis: Hours studied vs Exam score.
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

np.random.seed(7)
hours = np.round(np.random.uniform(1, 10, 20), 1)
scores = 50 + 5*hours + np.random.normal(0, 5, 20)  # linear relation + noise
df = pd.DataFrame({"hours": hours, "score": scores})

print("Correlation:", df.corr(numeric_only=True))

# Scatter + regression line
plt.figure(figsize=(6,4))
sns.regplot(x="hours", y="score", data=df, scatter_kws={"alpha": 0.7})
plt.title("Study Hours vs Exam Score")
plt.show()

# Optionally compute linear fit parameters
m, b = np.polyfit(df["hours"], df["score"], 1)
print(f"Fit: score ≈ {m:.2f} * hours + {b:.2f}")
